72  Fp64s& operator += (Fp64s r)
73  {
74  // Optimised for when q is the same for both
75  // operators (first if statement).
76  if(q == r.q)
77  {
78  // First number smaller
79  rawVal = rawVal + r.rawVal;
80  // No need to change Q, both are the same
81  }
82  else if(q > r.q)
83  {
84  // Second number has smaller Q, so result is in that precision
85  rawVal = (rawVal >> (q - r.q)) + r.rawVal;
86  // Change Q
87  q = r.q;
