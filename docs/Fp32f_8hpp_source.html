88 inline int32_t fixdiv(int32_t a, int32_t b)
89 {
90  #if 0
91  return (int32_t)((((int64_t)a) << q) / b);
92  #else
93  // The following produces the same results as the above but gcc 4.0.3
94  // generates fewer instructions (at least on the ARM processor).

100  };
101  } x;

109 namespace detail {
110  inline uint32_t CountLeadingZeros(uint32_t x)
111  {
112  uint32_t exp = 31;
113 
114  if (x & 0xffff0000) {
115  exp -= 16;
116  x >>= 16;
117  }
118 
119  if (x & 0xff00) {
120  exp -= 8;
121  x >>= 8;
122  }
123 
124  if (x & 0xf0) {
125  exp -= 4;
126  x >>= 4;
127  }
128 
129  if (x & 0xc) {
130  exp -= 2;
131  x >>= 2;
132  }
133 
134  if (x & 0x2) {
135  exp -= 1;
136  }
137 
138  return exp;
139  }
140 }

142 // q is the precision of the input
143 // output has 32-q bits of fraction
144 template <uint8_t q>
145 inline int32_t fixinv(int32_t a)
146 {
147  int32_t x;
148 
149  bool sign = false;
150 
151  if (a < 0) {
152  sign = true;
153  a = -a;
154  }
155 
156  static const uint16_t rcp_tab[] = {
157  0x8000, 0x71c7, 0x6666, 0x5d17, 0x5555, 0x4ec4, 0x4924, 0x4444
158  };
159 
160  int32_t exp = detail::CountLeadingZeros(a);
161  x = ((int32_t)rcp_tab[(a>>(28-exp))&0x7]) << 2;
162  exp -= 16;
163 
164  if (exp <= 0)
165  x >>= -exp;
166  else
167  x <<= exp;
168 
169  // Two iterations of newton-raphson x = x(2-ax)
170  x = FixMul<(32-q)>(x,((2<<(32-q)) - FixMul<q>(a,x)));
171  x = FixMul<(32-q)>(x,((2<<(32-q)) - FixMul<q>(a,x)));
172 
173  if (sign)
174  return -x;
175  else
176  return x;
177 }

581 // The multiply accumulate case can be optimized.
582 template <uint8_t q>
583 inline Fp32f<q> multiply_accumulate(
584  int32_t count,
585  const Fp32f<q> *a,
586  const Fp32f<q> *b)
587 {
588  long long result = 0;
589  for (int32_t i = 0; i < count; ++i)
590  result += static_cast<long long>(a[i].rawVal) * b[i].rawVal;
591  Fp32f<q> r;
592  r.rawVal = static_cast<int32_t>(result >> q);
593  return r;
